Example Case

A researcher is interested in how variables, such as GRE (Graduate Record Exam scores), GPA (grade point average) and prestige of the undergraduate institution, effect admission into graduate school. The response variable, admit/dont admit, is a binary variable.

Fitting the Model

mydata$rank <- factor(mydata$rank)
mylogit <- glm(admit ~ gre + gpa + rank, data = mydata, family = "binomial")

mydata$rankP <- predict(mylogit, newdata = mydata, type = "response")

ggplot( mydata, aes(x=gre, y=rankP)) +
  geom_point(aes(color = mydata$admit)) +
  geom_smooth(method = "glm", 
              method.args = list(family = "binomial"), 
              se = FALSE) 

ggplot( mydata, aes(x=gpa, y=rankP)) +
  geom_point(aes(color = mydata$admit)) +
  geom_smooth(method = "glm", 
              method.args = list(family = "binomial"), 
              se = FALSE)

library(forcats)
mydata$rank <- fct_rev(mydata$rank)
ggplot( mydata, aes(x=rank, y=rankP)) +
  geom_point() +
  geom_smooth(method = "glm", 
              method.args = list(family = "binomial"), 
              se = FALSE)
